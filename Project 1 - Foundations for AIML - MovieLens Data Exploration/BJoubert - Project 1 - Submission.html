# Now we can continue to answer the question.
# Just recreate the DataFrame so that Python does not cry each time the cell is re-run.
movieRatings = pd.merge(data, item, how='inner', on='movie id')
# We only need 'movie title' and 'rating'.
movieRatings = movieRatings[['movie title', 'rating']]
# Get a list containing the movies with more than 100 ratings.
moviesWithMoreThan100Ratings = movieRatings.groupby('movie title').count()
moviesWithMoreThan100Ratings.rename(columns={'rating':'count'}, inplace=True)
moviesWithMoreThan100Ratings = moviesWithMoreThan100Ratings[moviesWithMoreThan100Ratings['count'] >= 100]
moviesWithMoreThan100Ratings.reset_index(inplace=True)
moviesWithMoreThan100RatingsList = list(moviesWithMoreThan100Ratings['movie title'])
# Get the average ratings of each of these movies.
movieRatings.set_index('movie title', inplace=True)
movieRatings = movieRatings.loc[moviesWithMoreThan100RatingsList].groupby('movie title').mean().sort_values(by='rating', ascending=False)
# The first 25 movies are the 25 highest rated, since they hav already been sorted.
top25Movies = movieRatings.iloc[0:25]
